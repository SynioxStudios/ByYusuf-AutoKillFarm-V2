local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer
local RS = game:GetService("RunService")
local HttpService = game:GetService("HttpService")
local TeleportService = game:GetService("TeleportService")

_G.KillActive = true
_G.HopInterval = 50
local SecretAreaPos = CFrame.new(1947, 2, 6191)

task.spawn(function()
    local lighting = game:GetService("Lighting")
    local terrain = game:GetService("Workspace"):FindFirstChildOfClass('Terrain')
    lighting.Brightness = 0.5
    lighting.OutdoorAmbient = Color3.fromRGB(120, 120, 120)
    lighting.Ambient = Color3.fromRGB(120, 120, 120)
    lighting.ClockTime = 14
    if terrain then
        terrain.WaterWaveSize = 0
        terrain.WaterWaveSpeed = 0
        terrain.WaterReflectance = 0
    end
    for _, v in pairs(game:GetDescendants()) do
        if v:IsA("ParticleEmitter") or v:IsA("Trail") or v:IsA("BlurEffect") or v:IsA("SunRaysEffect") then
            v.Enabled = false
        end
    end
end)

task.spawn(function()
    local function setupCharacter(char)
        local hrp = char:WaitForChild("HumanoidRootPart")
        hrp.CFrame = SecretAreaPos
        local bodyVelocity = Instance.new("BodyVelocity")
        bodyVelocity.MaxForce = Vector3.new(100000, 0, 100000)
        bodyVelocity.Velocity = Vector3.new(0, 0, 0)
        bodyVelocity.P = 1250
        bodyVelocity.Parent = hrp
    end
    if LocalPlayer.Character then setupCharacter(LocalPlayer.Character) end
    LocalPlayer.CharacterAdded:Connect(setupCharacter)
end)

task.spawn(function()
    while _G.KillActive do
        pcall(function()
            local char = LocalPlayer.Character
            local punch = LocalPlayer.Backpack:FindFirstChild("Punch") or (char and char:FindFirstChild("Punch"))
            if punch then
                if punch.Parent ~= char then punch.Parent = char end
                LocalPlayer.muscleEvent:FireServer("punch", "rightHand")
                LocalPlayer.muscleEvent:FireServer("punch", "leftHand")
                local rHand = char:FindFirstChild("RightHand")
                if rHand then
                    for _, target in ipairs(Players:GetPlayers()) do
                        if target ~= LocalPlayer and target.Character then
                            local tRoot = target.Character:FindFirstChild("HumanoidRootPart")
                            local tHum = target.Character:FindFirstChild("Humanoid")
                            if tRoot and tHum and tHum.Health > 0 then
                                tRoot.Size = Vector3.new(15, 15, 15)
                                tRoot.CanCollide = false
                                firetouchinterest(rHand, tRoot, 1)
                                firetouchinterest(rHand, tRoot, 0)
                            end
                        end
                    end
                end
            end
        end)
        task.wait(0.0001)
    end
end)

local function forceHop()
    while true do
        local success, result = pcall(function()
            return HttpService:JSONDecode(game:HttpGet("https://games.roblox.com/v1/games/"..game.PlaceId.."/servers/Public?sortOrder=Desc&limit=100")).data
        end)
        
        if success and result then
            for _, server in pairs(result) do
                if server.playing >= 18 and server.playing < server.maxPlayers and server.id ~= game.JobId then
                    TeleportService:TeleportToPlaceInstance(game.PlaceId, server.id, LocalPlayer)
                end
            end
        end
        task.wait(0.5) 
    end
end

task.spawn(function()
    local t = _G.HopInterval
    while t > 0 do
        _G.CurrentHop = t
        task.wait(1)
        t = t - 1
    end
    forceHop()
end)
