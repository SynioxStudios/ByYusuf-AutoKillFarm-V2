task.spawn(function()
    while _G.KillActive do
        pcall(function()
            local char = LocalPlayer.Character
            local rightHand = char and char:FindFirstChild("RightHand")
            local leftHand = char and char:FindFirstChild("LeftHand")
            local punch = LocalPlayer.Backpack:FindFirstChild("Punch") or (char and char:FindFirstChild("Punch"))

            if punch then
                if punch.Parent ~= char then punch.Parent = char end
                
                LocalPlayer.muscleEvent:FireServer("punch", "rightHand")
                LocalPlayer.muscleEvent:FireServer("punch", "leftHand")

                if rightHand and leftHand then
                    for _, target in ipairs(Players:GetPlayers()) do
                        if target ~= LocalPlayer and not playerWhitelist[target.Name] then
                            local tRoot = target.Character and target.Character:FindFirstChild("HumanoidRootPart")
                            if tRoot then
                                firetouchinterest(rightHand, tRoot, 1)
                                firetouchinterest(leftHand, tRoot, 1)
                                firetouchinterest(rightHand, tRoot, 0)
                                firetouchinterest(leftHand, tRoot, 0)
                            end
                        end
                    end
                end
            end
        end)
        task.wait(0.01)
    end
end)

local function forceHop()
    TimerLabel.Text = "TP..."
    local success, result = pcall(function()
        return HttpService:JSONDecode(game:HttpGet("https://games.roblox.com/v1/games/"..game.PlaceId.."/servers/Public?sortOrder=Desc&limit=100")).data
    end)
    if success and result then
        for _, server in pairs(result) do
            if server.playing < server.maxPlayers and server.id ~= game.JobId then
                TeleportService:TeleportToPlaceInstance(game.PlaceId, server.id, LocalPlayer)
                task.wait(1.5)
            end
        end
    end
    task.wait(1); forceHop()
end

task.spawn(function()
    local t = _G.HopInterval
    while t > 0 do
        TimerLabel.Text = "HOP: "..t.."s"; task.wait(1); t = t - 1
    end
    forceHop()
end)
